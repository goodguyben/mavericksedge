    <script>
        let currentIndex = 0;
        const totalSlides = 5;
        let isScrolling = false;

        // Create floating particles
        function createParticles() {
            const particlesContainer = document.getElementById('particles');
            for (let i = 0; i < 20; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + '%';
                particle.style.animationDelay = Math.random() * 6 + 's';
                particle.style.animationDuration = (Math.random() * 3 + 3) + 's';
                particlesContainer.appendChild(particle);
            }
        }

        function updateSlide(index) {
            // Update images
            document.querySelectorAll('.image-card').forEach((card, i) => {
                card.classList.toggle('active', i === index);
                if (i === index) {
                    card.style.transform = 'translateZ(0) rotateY(0deg) scale(1)';
                    card.style.zIndex = 10;
                } else if (i < index) {
                    card.style.transform = `translateZ(-${(index - i) * 50}px) rotateY(-${(index - i) * 5}deg) scale(${1 - (index - i) * 0.05})`;
                    card.style.zIndex = 10 - (index - i);
                } else {
                    card.style.transform = `translateZ(-${(i - index) * 50}px) rotateY(${(i - index) * 5}deg) scale(${1 - (i - index) * 0.05})`;
                    card.style.zIndex = 10 - (i - index);
                }
            });

            // Update content
            document.querySelectorAll('.content-item').forEach((item, i) => {
                item.classList.toggle('active', i === index);
            });

            // Update progress dots
            document.querySelectorAll('.progress-dot').forEach((dot, i) => {
                dot.classList.toggle('active', i === index);
            });

            currentIndex = index;
        }

        function handleScroll() {
            if (isScrolling) return;
            
            const container = document.querySelector('.sticky-container');
            const rect = container.getBoundingClientRect();
            const progress = Math.max(0, Math.min(1, -rect.top / (rect.height - window.innerHeight)));
            const slideIndex = Math.floor(progress * totalSlides);
            const finalIndex = Math.min(slideIndex, totalSlides - 1);

            if (finalIndex !== currentIndex) {
                updateSlide(finalIndex);
            }
        }

        // Progress dot click handlers
        document.querySelectorAll('.progress-dot').forEach((dot, index) => {
            dot.addEventListener('click', () => {
                isScrolling = true;
                updateSlide(index);
                
                // Scroll to appropriate position
                const container = document.querySelector('.sticky-container');
                const targetScroll = container.offsetTop + (index / (totalSlides - 1)) * (container.offsetHeight - window.innerHeight);
                
                window.scrollTo({
                    top: targetScroll,
                    behavior: 'smooth'
                });

                setTimeout(() => {
                    isScrolling = false;
                }, 1000);
            });
        });

        // Auto-advance functionality (optional)
        let autoAdvanceInterval;
        
        function startAutoAdvance() {
            autoAdvanceInterval = setInterval(() => {
                if (!isScrolling) {
                    const nextIndex = (currentIndex + 1) % totalSlides;
                    const container = document.querySelector('.sticky-container');
                    const targetScroll = container.offsetTop + (nextIndex / (totalSlides - 1)) * (container.offsetHeight - window.innerHeight);
                    
                    window.scrollTo({
                        top: targetScroll,
                        behavior: 'smooth'
                    });
                }
            }, 4000);
        }

        function stopAutoAdvance() {
            clearInterval(autoAdvanceInterval);
        }

        // Event listeners
        window.addEventListener('scroll', handleScroll);
        window.addEventListener('resize', handleScroll);

        // Pause auto-advance on user interaction
        document.addEventListener('wheel', stopAutoAdvance);
        document.addEventListener('touchstart', stopAutoAdvance);
        document.addEventListener('keydown', stopAutoAdvance);

        // Initialize
        createParticles();
        updateSlide(0);
        
        // Start auto-advance after initial load
        setTimeout(startAutoAdvance, 3000);
    </script>
